Loops

Counting

  let i = 5;
  while (i > 0) {
    doStuff();
  }

  for (let i = 0; i < 5; i += 1) {
    doStuff();
  }

Enumeration

  const array = ['a', 'b', 'c'];
  for (const index in array) {
    console.log(index); // 0, 1, 2
    console.log(array[index]); // 'a', 'b', 'c'
  }

  const array = ['a', 'b', 'c'];
  for (const item of array) {
    console.log(item); // 'a', 'b', 'c'
  }

Recursion

  function countdown(n) {
    console.log(n);
    if (n > 0) {
      countdown(n - 1);
    }
  }

Scheduling

  setInterval(
    () => { console.log('gets called each second'); },
    1000
  );

  const countdown = (n) => {
    console.log(n);
    if (n > 0) {
      setTimeout(
        () => countdown(n - 1),
        1000
      );
    }
  };
